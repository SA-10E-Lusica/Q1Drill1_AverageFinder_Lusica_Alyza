from js import document, console from pyodide.ffi import create_proxy def compute_average(event): try: # Get input values from user score1 = float(document.getElementById("score1").value) score2 = float(document.getElementById("score2").value) # Compute and display average and pass/fail result average = (score1 + score2) / 2 avg_display = document.getElementById("average") result_display = document.getElementById("result") avg_display.innerText = f"{average:.2f}" if average >= 75: result_display.innerText = "✅ Passed" result_display.style.color = "limegreen" else: result_display.innerText = "❌ Failed" result_display.style.color = "red" except Exception as e: console.log(f"Error: {e}") document.getElementById("result").innerText = "⚠️ Please enter valid numbers" document.getElementById("result").style.color = "red" proxy_func = create_proxy(compute_average) document.getElementById("computeBtn").addEventListener("click", proxy_func)
